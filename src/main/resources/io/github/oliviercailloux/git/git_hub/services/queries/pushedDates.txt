query pushedDates($repositoryOwner: String!, $repositoryName: String!) {
  repository(owner: $repositoryOwner, name: $repositoryName) {
    url
    sshUrl
    id
    name
    createdAt
    owner {
      login
      url
    }
    heads: refs(refPrefix: "refs/heads/", first: 100) {
      pageInfo {
        hasNextPage
      }
      nodes {
        name
        prefix
        target {
          oid
          ... on Commit {
            ... commitHistory
          }
        }
      }
    }
    tags: refs(refPrefix: "refs/tags/", first: 100) {
      pageInfo {
        hasNextPage
      }
      nodes {
        name
        prefix
        target {
          oid
          ... on Tag {
            target {
              oid
              ... on Commit {
                ... commitHistory
              }
            }
	      }
        }
      }
    }
  }
}
