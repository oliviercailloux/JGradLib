query pushedDatesByOid($repositoryOwner: String!, $repositoryName: String!, $oid: GitObjectID!) {
  repository(owner: $repositoryOwner, name: $repositoryName) {
    object(oid: $oid) {
      ... on Commit {
        history {
          nodes {
            oid
            parents(first: 100) {
              totalCount
              nodes {
                oid
              }
            }
            authoredDate
            committedDate
            pushedDate
          }
        }
      }
    }
  }
}
