fragment repositoryWithIssuesWithHistory on Repository {
	url
	sshUrl
	id
	name
	createdAt
	owner {
	  login
	  url
	}
	masterObject: object(expression: "master:") {
		__typename
		... on Tree{
			entries{
				name
				type
			}
		}
	}
	pomObject: object(expression: "master:pom.xml") {
		__typename
		commitResourcePath
		commitUrl
	}
	issues(orderBy: {field: CREATED_AT, direction: ASC}, states: [OPEN, CLOSED], first: 100) {
	  totalCount
	  pageInfo {
	    hasNextPage
	  }
	  nodes {
	    title
	    url
	    repository {
	    	homepageUrl
	    	name
	    	owner
	    }
	    number
	    createdAt
	    closed
	    timeline(first: 100) {
	      totalCount
	      pageInfo {
	        hasNextPage
	      }
	      nodes {
	        __typename
	        ... on AssignedEvent {
	          createdAt
	          user {
	            login
	            url
	          }
	        }
	        ... on ClosedEvent {
	          createdAt
	        }
	        ... on ReopenedEvent {
	          createdAt
	        }
	        ... on RenamedTitleEvent {
	          createdAt
	          previousTitle
	          currentTitle
	        }
	        ... on UnassignedEvent {
	          createdAt
	          user {
	            login
	            url
	          }
	        }
	      }
	    }
	  }
	}
}
