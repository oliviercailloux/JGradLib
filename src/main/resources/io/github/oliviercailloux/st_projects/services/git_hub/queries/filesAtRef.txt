query filesAtRef($repositoryName: String!, $repositoryOwner: String!, $ref: String!) {
  repository(name: $repositoryName, owner: $repositoryOwner) {
	url
	sshUrl
	id
	name
	createdAt
	owner {
	  login
	  url
	}
	masterTree: object(expression: "master:") {
		__typename
		oid
		commitUrl
		... on Tree{
			entries{
				name
				type
				object {
					__typename
					oid
					commitUrl
				}
			}
		}
	}
	refTree: object(expression: $ref) {
		__typename
		... on Tree{
			entries{
				name
				type
			}
		}
	}
  }
}
